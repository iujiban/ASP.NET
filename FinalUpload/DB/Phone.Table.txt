CREATE TABLE PHONE_DATA (
	PHONE_KEY     NUMBER NOT NULL, -- PHONE_DATA SEQUENCE
	USER_KEY      NUMBER NOT NULL, -- 사용자 키 값
	UC_KEY        NUMBER, -- USER_CHECK 키 값
	RECIPIENT     VARCHAR2(256), -- 수신 인
	RNUMBER       VARCHAR2(256), -- 수신 번호
	CALLER        VARCHAR2(256), -- 발신 인
	CNUMBER       VARCHAR2(256), -- 발신 번호
	BUSINESSNAME  VARCHAR2(256), -- 거래처 명
	SERVICELEVEL  VARCHAR2(256), -- 서비스 레벨
	RECEIVEDATE   DATE, -- 통화 시작 날짜
	ENDDATE       DATE, -- 통화 종료 날짜
	OFFTIME       VARCHAR2(256), -- 업무 외 시간
	DURATIONCALL  NUMBER, -- 통화 시간
	CALLSTATUS    NVARCHAR2(256),  -- 통화 상태 (발신, 수신, 부재중)
	RESULTCALL    NVARCHAR2(256), -- 실제 통화 상태 (무시 해도 되는 데이터)
	ETC_1         DATE, 
	ETC_2         DATE,
	ETC_3         NUMBER,
	ETC_4         NUMBER,
	EMAIL         VARCHAR2(256), -- 사용자 USERKEY값의 따른 이메일
	NOTE          VARCHAR2(256)); -- 비고 (Edit 값)

ALTER TABLE PHONE_DATA ADD (
	CONSTRAINT PHONEDATA_PK PRIMARY KEY (PHONE_KEY));


CREATE TABLE PHONE_DATA_REPORT (
	PHONE_KEY     NUMBER NOT NULL, -- PHONE_DATA 키 값
	USER_KEY      NUMBER NOT NULL, -- 사용자 키 값
	UC_KEY        NUMBER, -- USER_CHECK KEY 값
	RECIPIENT     VARCHAR2(256), -- 수신 인
	RNUMBER       VARCHAR2(256), -- 수신 번호
	CALLER        VARCHAR2(256), -- 발신 인
	CNUMBER       VARCHAR2(256), -- 발신 번호
	BUSINESSNAME  VARCHAR2(256), -- 거래처 명
	SERVICELEVEL  VARCHAR2(256), -- 서비스레벨
	RECEIVEDATE   DATE, -- 통화 시작 날짜
	ENDDATE       DATE, -- 통화 종료 날짜
	OFFTIME       VARCHAR2(256), -- 업무 외 시간
	DURATIONCALL  NUMBER, -- 통화 시간
	CALLSTATUS    NVARCHAR2(256), -- 통화 상태
	RESULTCALL    NVARCHAR2(256), -- 실제 통화 상태 (무시 해도 되는 데이터)
	ETC_1         DATE,
	ETC_2         DATE,
	COSTCALL      NUMBER, -- 예상 정산 비용 1초당 3원
	ETC_4         NUMBER,
	EMAIL         VARCHAR2(256), -- 사용자 USERKEY값의 따른 이메일
	NOTE          VARCHAR2(256)); -- 비고

ALTER TABLE PHONE_DATA_REPORT ADD (
	CONSTRAINT PHONE_DATA_REPORT_UK UNIQUE (PHONE_KEY));

CREATE TABLE USER_CHECK (
	UC_KEY       NUMBER NOT NULL, -- USER_CHECK SEQUENCE
	USER_KEY     NUMBER NOT NULL, -- 사용자 키
	TARGET_DATE  DATE, -- 업로드 할 때 통화 월 달
	INSERT_DATE  DATE, -- 파일이 들어갈 때의 현재 시간
	DEL_DATE     DATE); -- 파일을 지웠을 때의 현재 시간

ALTER TABLE USER_CHECK ADD (
	CONSTRAINT USER_CHECK_PK PRIMARY KEY (UC_KEY));

CREATE TABLE VACATION (
	VNUMBER       NUMBER NOT NULL, -- VACATION SEQUENCE
	VACATIONDATE  DATE); -- 법정 공휴일의 날짜

ALTER TABLE VACATION ADD (
	CONSTRAINT VACATION_PK PRIMARY KEY (VNUMBER));
